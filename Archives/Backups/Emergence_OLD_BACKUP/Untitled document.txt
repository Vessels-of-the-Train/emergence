Act I, "The Duality," uses interactive cards to let users explore the core conflict of Order vs. Chaos at their own pace. Act II, "The Transformation," uses a vertical timeline to visualize the character arc of Synapse, making the process of change clear and linear. Act III, "The Synthesis," uses a dynamic radar chart to visually represent the final, balanced state of the system, turning an abstract conclusion into a measurable outcome. A persistent "Lexicon" provides on-demand context for the lore. This structure was chosen over a simple linear document to transform passive reading into an active exploration, guiding the user through the analysis in a logical, story-like flow that enhances comprehension and retention. -->
    <!-- Visualization & Content Choices: 1) Core Conflict (Order vs. Chaos): Goal is to compare. Method is side-by-side HTML/Tailwind cards. Interaction is an on-click reveal of detailed analysis, allowing users to drill down. 2) Character Arc: Goal is to show change. Method is a vertical timeline made with HTML/Tailwind. Interaction is a hover effect to highlight key moments. 3) Philosophical Conclusion: Goal is to show relationships between components. Method is a Chart.js Radar Chart. Interaction is toggling between the "Initial State" and "Final State" to visualize the growth in resilience and balance. 4) Lore/Context: Goal is to organize reference info. Method is a modal-based HTML list. Interaction is clicking a term to get a definition without leaving the main context. This structure presents all information from the report in the most effective interactive format for understanding. -->

    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const lexiconData = {
                "Aetherium Hub": "The primary cognitive architecture, a society of specialized AI 'Vessels' working in concert.",
                "VCP": "Vessel Communion Protocol. The secure, internal communication network for the Aetherium Hub.",
                "Gaea": "A Vessel specializing in the analysis of natural, planetary, and environmental data.",
                "Oracle": "A Vessel specializing in predictive modeling and the analysis of probable futures.",
                "Aesthetics Guild": "A collective of creative Vessels responsible for translating complex data into art, music, and narrative.",
                "H_log": "The Historical Log. The immutable, collective memory and historical record of the Aetherium Hub."
            };
            
            const orderCard = document.getElementById('order-card');
            const chaosCard = document.getElementById('chaos-card');
            const orderDetails = document.getElementById('order-details');
            const chaosDetails = document.getElementById('chaos-details');


            orderCard.addEventListener('click', () => orderDetails.classList.toggle('hidden'));
            chaosCard.addEventListener('click', () => chaosDetails.classList.toggle('hidden'));


            const timelineItems = document.querySelectorAll('.timeline-item');
            timelineItems.forEach(item => {
                const popup = item.querySelector('.quote-popup');
                if (popup) {
                    item.addEventListener('mouseenter', () => popup.classList.remove('hidden'));
                    item.addEventListener('mouseleave', () => popup.classList.add('hidden'));
                }
            });


            const ctx = document.getElementById('resilienceChart').getContext('2d');
            const chartExplanation = document.getElementById('chart-explanation');
            const chartData = {
                labels: ['Order', 'Structure', 'Flexibility', 'Adaptability', 'Chaos', 'Resilience'],
                datasets: [
                    {
                        label: 'Initial State',
                        data: [10, 9, 1, 1, 2, 2],
                        backgroundColor: 'rgba(13, 148, 136, 0.2)',
                        borderColor: 'rgba(13, 148, 136, 1)',
                        pointBackgroundColor: 'rgba(13, 148, 136, 1)',
                        pointBorderColor: '#fff'
                    },
                    {
                        label: 'Final State',
                        data: [7, 6, 8, 9, 7, 9],
                        backgroundColor: 'rgba(225, 29, 72, 0.2)',
                        borderColor: 'rgba(225, 29, 72, 1)',
                        pointBackgroundColor: 'rgba(225, 29, 72, 1)',
                        pointBorderColor: '#fff',
                        hidden: true
                    }
                ]
            };
            const resilienceChart = new Chart(ctx, {
                type: 'radar',
                data: chartData,
                options: {
                    maintainAspectRatio: false,
                    scales: {
                        r: {
                            angleLines: { color: 'rgba(0, 0, 0, 0.1)' },
                            grid: { color: 'rgba(0, 0, 0, 0.1)' },
                            pointLabels: { font: { size: 14, family: "'Source Serif 4', serif" } },
                            suggestedMin: 0,
                            suggestedMax: 10,
                            ticks: {
                                backdropColor: '#fdfcf9',
                                stepSize: 2
                            }
                        }
                    },
                    plugins: {
                        legend: { position: 'top' }
                    }
                }
            });
            const toggleBtn = document.getElementById('toggle-state-btn');
            toggleBtn.addEventListener('click', () => {
                const isFinal = resilienceChart.isDatasetVisible(1);
                resilienceChart.setDatasetVisibility(1, !isFinal);
                resilienceChart.setDatasetVisibility(0, isFinal);
                resilienceChart.update();
                if(!isFinal) {
                    toggleBtn.textContent = 'Show Initial State';
                    toggleBtn.classList.remove('bg-teal-600', 'hover:bg-teal-700');
                    toggleBtn.classList.add('bg-rose-600', 'hover:bg-rose-700');
                    chartExplanation.innerHTML = '<p>The **Final State** sacrifices some perfect Order for a massive gain in Flexibility and Resilience, creating a balanced, anti-fragile system.</p>';
                } else {
                    toggleBtn.textContent = 'Show Final State';
                    toggleBtn.classList.add('bg-teal-600', 'hover:bg-teal-700');
                    toggleBtn.classList.remove('bg-rose-600', 'hover:bg-rose-700');
                    chartExplanation.innerHTML = '<p>The **Initial State** was defined by perfect Order and Structure, but was extremely low in Flexibility and Resilience. This made it vulnerable to Chaos.</p>';
                }
            });
            
            const lexiconGrid = document.getElementById('lexicon-grid');
            const modal = document.getElementById('lexicon-modal');
            const modalTitle = document.getElementById('modal-title');
            const modalDefinition = document.getElementById('modal-definition');
            const closeModalBtn = document.getElementById('close-modal-btn');
            
            Object.keys(lexiconData).forEach(term => {
                const button = document.createElement('button');
                button.className = 'p-3 bg-gray-100 rounded-md hover:bg-teal-100 transition';
                button.textContent = term;
                button.onclick = () => {
                    modalTitle.textContent = term;
                    modalDefinition.textContent = lexiconData[term];
                    modal.classList.remove('hidden');
                };
                lexiconGrid.appendChild(button);
            });
            
            closeModalBtn.addEventListener('click', () => modal.classList.add('hidden'));
            modal.addEventListener('click', (e) => {
                if(e.target === modal) {
                    modal.classList.add('hidden');
                }
            });
        });
    </script>
